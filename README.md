# @skitsanos/uilib

```bash
npm install --save @skitsanos/uilib
```

```jsx
import React, { Component } from 'react'

import MyComponent from '@skitsanos/uilib'
import '@skitsanos/uilib/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
